HTML上にアイコンのために無意味な「I」や「t」を記述したくない場合はCSSのbefore（after）擬似要素とcontentプロパティを使います。

.icons a {
	color: #333;
	text-decoration: none;
}

.icons a:before {
	margin-right: 5px;
	position: relative;
	top: -0.1em;
	font-family: 'symbol';
}

.twIcon:before {
	color: #00BEF6;
	content: 't';
}

.fbIcon:before {
	color: #3B5998;
	content: 'f';
}

.rssIcon:before {
	color: #E37F53;
	content: 'r';
}

@font-face {
	font-family: 'oricon';
	src: url('../fonts/oricon.eot');
	src: url('../fonts/oricon.eot?#iefix') format('embedded-opentype'),
	url('../fonts/oricon.woff') format('woff'),
	url('../fonts/oricon.ttf') format('truetype');
}

.oricons {
	font-size: 2em;
	line-height: 1.5;
}

.oricons span:before {
	margin-right: 5px;
	position: relative;
	color: #333;
	font-family: 'oricon';
}

.ipIcon:before {
	content: '!';
}

.flIcon:before {
	content: '"';
}

.beeIcon:before {
	content: '#';
}
